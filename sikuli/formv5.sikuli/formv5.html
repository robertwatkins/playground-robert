#Because the base language we are using is python, we have
#all the language capabilities we need to make a more 
#fully-featured script.
#
#In this case, I've added a loop. Note that all the steps
#to be performed in the loop are indented. This is how
#python knows how many lines to repeat.
i=0
while i<2:
    i=i+1
    type(Pattern().targetOffset(-93,20), "01/01/2017")
    type(Pattern().targetOffset(-69,21), "70\t\t")
    click(Pattern().targetOffset(-16,-6))
    type(Pattern().targetOffset(-76,28), "1\t\t")
    type(Pattern().targetOffset(-86,21),"2\t\t" )
    type(Pattern().targetOffset(-104,18), "3\t\t")
    type(Pattern().targetOffset(-120,20), "4\t\t")
    type(Pattern().targetOffset(-52,25), "5\t\t")
    type(Pattern().similar(0.83).targetOffset(-91,22), "6\t\t")
    type(Pattern().targetOffset(-95,21), "7\t\t")
    type(Pattern().targetOffset(-86,23), "8\t\t")
    click()
    wait()
    click()
    wait()
